69 size_t put(const KeyValue& kv) {
70 size_t entry_size = kv.key.size() + kv.value.size() + sizeof(uint64_t) + sizeof(bool);
71
72 // Check if key already exists to avoid double-counting
73 for (size_t i = 0; i < entries.size(); i++) {
74 if (entries[i].key == kv.key) {
75 // Subtract the size of the existing entry
76 size_bytes -= entries[i].key.size() + entries[i].value.size() +
77 sizeof(uint64_t) + sizeof(bool);
78 // Replace the entry
79 entries[i] = kv;
80 size_bytes += entry_size;
81 return entry_size;
82 }
83 }
84
85 // Key doesn't exist, add it
86 entries.push_back(kv);
87 size_bytes += entry_size;
88
89 // Keep entries sorted by key
90 sort(entries.begin(), entries.end(),
91 [](const KeyValue& a, const KeyValue& b) { return a.key < b.key; });
92
93 return entry_size;
94 }
95
101 const KeyValue* get(const string& key) const {
102 // Binary search since entries are sorted
103 auto it = lower_bound(entries.begin(), entries.end(), key,
104 [](const KeyValue& kv, const string& k) { return kv.key < k; });
105
106 if (it != entries.end() && it->key == key) {
107 return &(*it);
108 }
109 return nullptr;
110 }
