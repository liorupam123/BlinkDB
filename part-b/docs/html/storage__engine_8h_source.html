49 BloomFilter(size_t expected_items = 10000, double false_positive_rate = 0.01) {
50 bit_array_size_ = static_cast<size_t>(-expected_items * std::log(false_positive_rate) / (std::log(2) * std::log(2)));
51 num_hashes_ = static_cast<size_t>(bit_array_size_ * std::log(2) / expected_items);

54 num_hashes_ = std::max(num_hashes_, size_t(2));
55 num_hashes_ = std::min(num_hashes_, size_t(20));

64 void add(const std::string& key) {
65 size_t h1 = hash1(key);
66 size_t h2 = hash2(key);
67
68 for (size_t i = 0; i < num_hashes_; ++i) {
69 size_t hash = (h1 + i * h2) % bit_array_size_;

71 }

79 bool mightContain(const std::string& key) const {
80 size_t h1 = hash1(key);
81 size_t h2 = hash2(key);
82
83 for (size_t i = 0; i < num_hashes_; ++i) {
84 size_t hash = (h1 + i * h2) % bit_array_size_;
85 if (!bits_[hash]) {
86 return false;
87 }
88 }
89
90 return true;
91 }

98 size_t hash1(const std::string& key) const {
99 size_t hash = 14695981039346656037ULL;
100 for (char c : key) {
101 hash ^= static_cast<size_t>(c);
102 hash *= 1099511628211ULL;
103 }
104 return hash;

176 cache_map_.erase(last.first);
177 cache_list_.pop_back();
178 }
179
180 // Insert new item at front
181 cache_list_.emplace_front(key, value);
182 cache_map_[key] = cache_list_.begin();

431 void addKey(const std::string& key, size_t& common_prefix_len) {
432 if (prefix.empty()) {
433 prefix = key;
434 common_prefix_len = key.length();
435 return;
436 }
437
438 // Find common prefix length
439 size_t min_len = std::min(prefix.length(), key.length());
440 common_prefix_len = 0;
441
442 while (common_prefix_len < min_len && prefix[common_prefix_len] == key[common_prefix_len]) {
443 common_prefix_len++;
444 }
445
446 // Update prefix
447 if (common_prefix_len < prefix.length()) {
448 prefix = prefix.substr(0, common_prefix_len);
449 }
450
451 // Add suffix
452 std::string suffix = key.substr(common_prefix_len);
453 suffix_to_index[suffix] = suffix_to_index.size();
454 }

457 if (prefix.empty() || key.length() < prefix.length()) {
458 return key;
459 }
460
461 // Check if key starts with prefix
462 for (size_t i = 0; i < prefix.length(); i++) {
463 if (key[i] != prefix[i]) {
464 return key;
465 }
466 }
467
468 return key.substr(prefix.length());

592 explicit SSTable(const std::map<std::string, std::string>& source_data, size_t expected_items = 10000)
593 : bloom_filter(expected_items) {
594
595 if (!source_data.empty()) {
596 data = source_data;
597 min_key = source_data.begin()->first;
598 max_key = source_data.rbegin()->first;
599
600 // Build fence pointers and optimize key prefixes

610 bool get(const std::string& key, std::string& value) const {
611 // Increment access count for this SSTable - now works with mutable atomic
612 access_count++;
613
614 // Use bloom filter for fast negative lookups
615 if (!bloom_filter.mightContain(key)) {
616 return false;
617 }
618
619 // Use key range for quick rejection
620 if (!min_key.empty() && !max_key.empty()) {
621 if (key < min_key || key > max_key) {
622 return false;
623 }
624 }
625
626 // Actual lookup
627 auto it = data.find(key);
628 if (it == data.end()) {
629 return false;
630 }
631
632 if (it->second.empty()) {
633 // Tombstone value
634 return false;
635 }
636
637 value = it->second;
638 return true;
639 }

641 bool mightContain(const std::string& key) const {
642 // Key range check for quick rejection
643 if (!min_key.empty() && !max_key.empty()) {
644 if (key < min_key || key > max_key) {
645 return false;
646 }
647 }
648
649 return bloom_filter.mightContain(key);

652 size_t memoryUsage() const {
653 size_t usage = 0;

655 // Size of keys and values
656 for (const auto& [key, value] : data) {
657 usage += key.size() + value.size() + sizeof(key) + sizeof(value);

696 // Fast hash function for partitioning
697 uint32_t murmurHash(const std::string& key) const {
698 const uint32_t m = 0x5bd1e995;
699 const int r = 24;
700 uint32_t h = 0; // Seed
701
702 // Mix 4 bytes at a time into the hash
703 const unsigned char* data = (const unsigned char*)key.data();
704 size_t len = key.size();
705
706 while (len >= 4) {
707 uint32_t k = *(uint32_t*)data;
708
709 k *= m;
710 k ^= k >> r;
711 k *= m;
712
713 h *= m;
714 h ^= k;
715
716 data += 4;
717 len -= 4;
718 }
719
720 // Handle the last few bytes
721 switch (len) {
722 case 3: h ^= data[2] << 16; // FALLTHROUGH
723 case 2: h ^= data[1] << 8; // FALLTHROUGH
724 case 1: h ^= data[0];
725 h *= m;
726 };
727
728 // Do a few final mixes
729 h ^= h >> 13;
730 h *= m;
731 h ^= h >> 15;
732
733 return h;
734 }
